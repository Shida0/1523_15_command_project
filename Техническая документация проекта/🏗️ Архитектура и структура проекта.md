### Файловая структура

``` text
asteroid-watch/
├── 📁 models/                   # Модели базы данных
│   ├── __init__.py
│   ├── base.py                  # Базовый класс моделей
│   ├── asteroid.py              # Модель астероидов
│   ├── close_approach.py        # Модель сближений
│   └── threat_assessment.py
│
├── 📁 controllers/              # Контроллеры доступа к данным
│   ├── __init__.py
│   ├── base_controller.py       # Базовый контроллер CRUD
│   ├── asteroid_controller.py   # Контроллер астероидов
│   ├── approach_controller.py   # Контроллер сближений
│   └── threat_controller.py     # Контроллер оценок угроз
|
├── 📁 database/                 # создание БД
|   ├── __init__.py
│   ├── engine.py                # Асинхронный движок
│   └── get_db_config.py         # Модель оценок угроз
|
├── 📁 utils/                    #   утилиты для кода
|    ├── __init__.py
│   ├── get_date.py              # Парсинг дат (предполагаемый)
│   └── space_math.py            # Математические расчеты (предполагаемый)
│
├── 📁 external_apis/            # Клиенты NASA API
│   ├── __init__.py
│   ├── sbdb_api.py              # Клиент SBDB API
│   ├── cad_api.py               # Клиент CAD API
│   └── sentry_api.py            # Клиент Sentry API
│
├── 📁 api_functions/            # функции-обёртки для работы с внешнимми API
│   ├── __init__.py
│   ├── get_data.py              # Уровень 1: получение астероидов
│   ├── get_approaches.py        # Уровень 2: получение сближений
│   └── get_treat.py             # Уровень 3: получение угроз
│
├── 📁 schemas/                  # Pydantic схемы валидации
│   ├── __init__.py
│   ├── base_schema.py           # Базовые схемы
│   ├── asteroid_schema.py       # Схемы астероидов
│   ├── approach_schema.py       # Схемы сближений
│   └── threat_schema.py         # Схемы оценок угроз
│
├── 📁 services/                 # Бизнес-логика (сервисы)
│   ├── __init__.py
│   ├── base_service.py          # Базовый класс с общими методами
|	├── asteroid_service.py      # Специализированный сервис для астероидов
|	├── approach_service.py      # Специализированный сервис для сближений
|	├── threat_service.py        # Специализированный сервис для угроз
│   └── update_service.py        # Сервис обновления данных
│
├── requirements.txt             # Зависимости Python
└── README.md                    # Основная документация
```
