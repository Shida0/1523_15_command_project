### Файловая структура

``` text
asteroid-watch/
├── 📁 models/                    # Модели базы данных
│   ├── __init__.py
│   ├── base.py                  # Базовый класс моделей
│   ├── asteroid.py              # Модель астероидов (PHA)
│   └── close_approach.py        # Модель сближений
│
├── 📁 controllers/              # Контроллеры (Data Access Layer)
│   ├── __init__.py
│   ├── base_controller.py       # Базовый контроллер с CRUD
│   ├── asteroid_controller.py   # Контроллер астероидов
│   └── approach_controller.py   # Контроллер сближений
│
├── 📁 external_apis/            # Клиенты NASA API
│   ├── __init__.py
│   ├── sbdb_api.py              # Клиент SBDB API (астероиды)
│   ├── cad_api.py               # Клиент CAD API (сближения)
│   ├── sentry_api.py            # Клиент Sentry API (оценка рисков)
│   ├── get_date.py              # Парсинг дат из CAD API
│   └── space_math.py            # Математические расчеты
│
├── 📁 utils/                    # Вспомогательные модули
│   ├── __init__.py
│   ├── get_data.py              # Уровень 1: Получение данных астероидов
│   ├── get_approaches.py        # Уровень 2: Получение данных о сближениях
│   └── get_treat.py             # Уровень 3: Получение оценок рисков
│
├── 📁 migrations/               # Миграции базы данных (Alembic)
├── .env                         # Пример переменных окружения
├── requirements.txt             # Зависимости Python
├── alembic.ini                  # Конфигурация Alembic
└── README.md                    # Эта документация
```
